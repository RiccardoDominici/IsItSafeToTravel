**For src/pages/en/index.astro:**

```astro
---
import Base from '../../layouts/Base.astro';
import SafetyMap from '../../components/SafetyMap.astro';
import { useTranslations } from '../../i18n/utils';
import type { Lang } from '../../i18n/ui';
import type { DailySnapshot } from '../../pipeline/types';

const lang: Lang = 'en';
const t = useTranslations(lang);

// Load score data at build time
// If latest.json doesn't exist yet (pipeline hasn't run), use empty array
let scores = [];
try {
  const fs = await import('node:fs');
  const path = await import('node:path');
  const dataPath = path.join(process.cwd(), 'data', 'scores', 'latest.json');
  if (fs.existsSync(dataPath)) {
    const snapshot: DailySnapshot = JSON.parse(fs.readFileSync(dataPath, 'utf-8'));
    scores = snapshot.countries;
  }
} catch {
  // Pipeline hasn't run yet, map will show all countries as unscored
}
---

<Base lang={lang}>
  <section class="flex flex-col" style="height: calc(100vh - 4rem);">
    <div class="text-center pt-4 pb-2 px-4">
      <h1 class="font-heading text-2xl md:text-3xl font-bold text-sand-800 dark:text-sand-100">
        {t('hero.title')}
      </h1>
      <p class="text-sm md:text-base text-sand-500 dark:text-sand-400 mt-1">
        {t('map.tagline')}
      </p>
    </div>
    <div class="flex-1 min-h-0">
      <SafetyMap lang={lang} scores={scores} />
    </div>
  </section>
</Base>
```
