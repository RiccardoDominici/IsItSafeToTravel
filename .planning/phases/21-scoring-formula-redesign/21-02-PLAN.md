```typescript
/**
 * Compute a freshness weight for an indicator based on its data age.
 * Uses exponential decay: weight = e^(-lambda * ageDays)
 * where lambda = ln(2) / halfLifeDays
 *
 * Returns 1.0 for fresh data, approaches 0.0 for stale data.
 * Clamped to 0 if age exceeds maxAgeDays.
 *
 * @param dataAgeMs - Age of data in milliseconds (Date.now() - dataDate)
 * @param halfLifeDays - Number of days until weight reaches 0.5
 * @param maxAgeDays - Maximum age in days; data older than this gets weight 0
 * @returns Weight between 0 and 1
 */
export function freshnessWeight(
  dataAgeMs: number,
  halfLifeDays: number,
  maxAgeDays: number,
): number {
  const ageDays = dataAgeMs / (1000 * 60 * 60 * 24);

  if (ageDays > maxAgeDays) return 0;
  if (ageDays <= 0) return 1;

  const lambda = Math.LN2 / halfLifeDays;
  return Math.exp(-lambda * ageDays);
}
```

Key changes to `computeCountryScore`:

1. **Add `sourcesConfig` parameter** — `computeCountryScore(iso3, allIndicators, weightsConfig, countryEntry, advisories, sources, sourcesConfig?: SourcesConfig)`. When `sourcesConfig` is undefined, fall back to legacy equal-averaging behavior for backward compatibility.

2. **Separate indicators by tier**: For each indicator, look up its `source` in `sourcesConfig.sources` to determine tier (baseline vs signal). Unknown sources default to baseline tier.

3. **Compute freshness weights**: For each indicator, compute age from `fetchedAt` or `dataDate` (fall back to 1.0 weight if neither is set — backward compat with existing data that has no timestamps). Call `freshnessWeight(ageMs, config.decayHalfLifeDays, config.maxAgeDays)`.

4. **Per-pillar scoring with sub-weights**: For each pillar:
   - Separate normalized indicators into baseline and signal groups
   - Apply per-indicator sub-weights from `pillarDef.indicatorWeights` (if not set, fall back to equal weights — backward compat)
   - Compute baseline tier score: sub-weight * freshnessWeight weighted average of baseline indicators
   - Compute signal tier score: sub-weight * freshnessWeight weighted average of signal indicators
   - Blend: `pillarScore = baselineScore * (1 - effectiveSignalInfluence) + signalScore * effectiveSignalInfluence`
   - Where `effectiveSignalInfluence = maxSignalInfluence * signalCompleteness`
   - `signalCompleteness = min(1, availableSignalIndicators / expectedSignalIndicators)` — use count of signal indicators defined in the pillar's `indicators` array

5. **Graceful degradation**: If no signal indicators have data for a country, `signalCompleteness = 0` and `effectiveSignalInfluence = 0`, so the score equals the pure baseline score. This is the FORM-04 requirement.

6. **Composite score unchanged**: `score = weightedSum * 9 + 1`, same as current.
