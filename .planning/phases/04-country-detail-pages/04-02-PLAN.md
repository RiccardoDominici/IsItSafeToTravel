**Create `src/pages/en/country/[slug].astro`:**

```astro
---
import type { GetStaticPaths } from 'astro';
import type { ScoredCountry } from '../../../pipeline/types';
import type { HistoryPoint } from '../../../lib/scores';
import Base from '../../../layouts/Base.astro';
import ScoreHero from '../../../components/country/ScoreHero.astro';
import PillarBreakdown from '../../../components/country/PillarBreakdown.astro';
import AdvisorySection from '../../../components/country/AdvisorySection.astro';
import TrendSparkline from '../../../components/country/TrendSparkline.astro';
import CountrySummary from '../../../components/country/CountrySummary.astro';
import SourcesList from '../../../components/country/SourcesList.astro';
import { loadLatestScores, loadHistoricalScores } from '../../../lib/scores';
import { useTranslations } from '../../../i18n/utils';
import type { Lang } from '../../../i18n/ui';

const lang: Lang = 'en';
const t = useTranslations(lang);

export const getStaticPaths: GetStaticPaths = async () => {
  const countries = loadLatestScores();
  const history = loadHistoricalScores(90);

  return countries.map((country) => ({
    params: { slug: country.iso3.toLowerCase() },
    props: {
      country,
      trend: history.get(country.iso3) ?? [],
    },
  }));
};

interface Props {
  country: ScoredCountry;
  trend: HistoryPoint[];
}

const { country, trend } = Astro.props;
---
<Base lang={lang} title={`${country.name[lang]} - ${t('country.score_label')}`} description={`${country.name[lang]} safety score: ${country.score}/10. Category breakdown, travel advisories, and trend data.`}>
  <div class="max-w-4xl mx-auto px-4 py-8 space-y-8">
    <nav class="text-sm">
      <a href={`/${lang}/`} class="text-terracotta-500 hover:text-terracotta-600 underline">{t('country.back_to_map')}</a>
    </nav>

    <ScoreHero country={country} lang={lang} />
    <PillarBreakdown pillars={country.pillars} lang={lang} />
    <AdvisorySection advisories={country.advisories} lang={lang} />
    <TrendSparkline data={trend} lang={lang} />
    <CountrySummary country={country} lang={lang} />
    <SourcesList sources={country.sources} lang={lang} />
  </div>
</Base>
```
