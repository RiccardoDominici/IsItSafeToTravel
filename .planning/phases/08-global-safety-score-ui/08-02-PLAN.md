```typescript
/**
 * Build JSON-LD structured data for the global safety page.
 * Uses WebPage schema with aggregateRating for the global score.
 */
export function buildGlobalSafetyJsonLd(
  globalScore: number,
  canonicalUrl: string,
  lang: Lang,
): Record<string, unknown> {
  return {
    '@context': 'https://schema.org',
    '@graph': [
      {
        '@type': 'WebPage',
        '@id': canonicalUrl,
        url: canonicalUrl,
        name: lang === 'en' ? 'Global Safety Score' : 'Punteggio di Sicurezza Globale',
        description: lang === 'en'
          ? `Current global safety score: ${globalScore.toFixed(1)}/10. Track world safety trends over time.`
          : `Punteggio di sicurezza globale attuale: ${globalScore.toFixed(1)}/10. Segui le tendenze di sicurezza mondiale nel tempo.`,
        inLanguage: lang === 'it' ? 'it-IT' : 'en-US',
      },
      {
        '@type': 'AggregateRating',
        ratingValue: globalScore.toFixed(1),
        bestRating: '10',
        worstRating: '1',
        ratingCount: 248,
        itemReviewed: {
          '@type': 'Place',
          name: lang === 'en' ? 'World' : 'Mondo',
        },
      },
    ],
  };
}
```

Frontmatter:
```astro
---
export const prerender = true;
import Base from '../../layouts/Base.astro';
import { useTranslations } from '../../i18n/utils';
import type { Lang } from '../../i18n/ui';
import { loadLatestSnapshot, loadGlobalHistory } from '../../lib/scores';
import { buildGlobalSafetyJsonLd } from '../../lib/seo';
import { scoreToColor } from '../../lib/colors';
import { scaleLinear, scaleTime, line, timeFormat } from 'd3';

const lang: Lang = 'en';
const t = useTranslations(lang);

const snapshot = loadLatestSnapshot();
const globalScore = snapshot?.globalScore ?? 0;
const bgColor = scoreToColor(globalScore);
const globalHistory = loadGlobalHistory();

const canonicalUrl = new URL(Astro.url.pathname, Astro.site).href;
const jsonLd = buildGlobalSafetyJsonLd(globalScore, canonicalUrl, lang);

// Build full-size trend chart SVG at build time
const chartWidth = 800;
const chartHeight = 300;
const margin = { top: 20, right: 20, bottom: 40, left: 50 };
const innerWidth = chartWidth - margin.left - margin.right;
const innerHeight = chartHeight - margin.top - margin.bottom;

let pathD = '';
let endpointCx = 0;
let endpointCy = 0;
const showChart = globalHistory.length >= 2;

// Axis tick data
let xTicks: Array<{ x: number; label: string }> = [];
let yTicks: Array<{ y: number; label: string }> = [];

if (showChart) {
  const dates = globalHistory.map(d => new Date(d.date));
  const x = scaleTime()
    .domain([dates[0], dates[dates.length - 1]])
    .range([0, innerWidth]);

  const y = scaleLinear()
    .domain([1, 10])
    .range([innerHeight, 0]);

  const lineGen = line<{ date: string; score: number }>()
    .x(d => x(new Date(d.date)))
    .y(d => y(d.score));

  pathD = lineGen(globalHistory) ?? '';

  const lastPoint = globalHistory[globalHistory.length - 1];
  endpointCx = x(new Date(lastPoint.date));
  endpointCy = y(lastPoint.score);

  // X axis ticks (5-7 evenly spaced dates)
  const tickCount = Math.min(6, dates.length);
  const fmt = timeFormat('%b %d');
  for (let i = 0; i < tickCount; i++) {
    const idx = Math.round((i / (tickCount - 1)) * (dates.length - 1));
    xTicks.push({ x: x(dates[idx]), label: fmt(dates[idx]) });
  }

  // Y axis ticks (scores 2, 4, 6, 8, 10)
  yTicks = [2, 4, 6, 8, 10].map(s => ({ y: y(s), label: String(s) }));
}
---
```
