```typescript
// --- Raw data types ---
export interface FetchResult {
  source: string;
  success: boolean;
  countriesFound: number;
  error?: string;
  fetchedAt: string; // ISO 8601
}

export interface RawIndicator {
  countryIso3: string;
  indicatorName: string;
  value: number;
  year: number;
  source: string;
}

export interface RawSourceData {
  source: string;
  fetchedAt: string;
  indicators: RawIndicator[];
}

// --- Scoring config ---
export type PillarName = 'conflict' | 'crime' | 'health' | 'governance' | 'environment';

export interface PillarWeight {
  name: PillarName;
  weight: number;
  indicators: string[];
}

export interface WeightsConfig {
  version: string;
  pillars: PillarWeight[];
}

// --- Scored output ---
export interface IndicatorScore {
  name: string;
  rawValue: number;
  normalizedValue: number; // 0-1
  source: string;
  year: number;
}

export interface PillarScore {
  name: PillarName;
  score: number; // 0-1
  weight: number;
  indicators: IndicatorScore[];
  dataCompleteness: number; // 0-1 percentage of available indicators
}

export interface AdvisoryInfo {
  level: number | string;
  text: string;
  source: string;
  url: string;
  updatedAt: string;
}

export interface ScoredCountry {
  iso3: string;
  name: { en: string; it: string };
  score: number; // 1-10, one decimal
  scoreDisplay: number; // 1-10 integer
  pillars: PillarScore[];
  advisories: {
    us?: AdvisoryInfo;
    uk?: AdvisoryInfo;
  };
  dataCompleteness: number; // 0-1 overall
  lastUpdated: string; // ISO 8601
  sources: SourceMeta[];
}

export interface SourceMeta {
  name: string;
  url: string;
  fetchedAt: string;
  description: string;
}

// --- Country mapping ---
export interface CountryEntry {
  iso3: string;
  iso2: string;
  name: { en: string; it: string };
}

// --- Daily snapshot ---
export interface DailySnapshot {
  date: string; // YYYY-MM-DD
  generatedAt: string; // ISO 8601
  pipelineVersion: string;
  weightsVersion: string;
  countries: ScoredCountry[];
  fetchResults: FetchResult[];
}
```

1. Create `src/pipeline/utils/fs.ts` with helper functions:
```typescript
import { mkdirSync, writeFileSync, readFileSync, existsSync } from 'node:fs';
import { join, dirname } from 'node:path';

export function ensureDir(dirPath: string): void {
  mkdirSync(dirPath, { recursive: true });
}

export function writeJson(filePath: string, data: unknown): void {
  ensureDir(dirname(filePath));
  writeFileSync(filePath, JSON.stringify(data, null, 2) + '\n', 'utf-8');
}

export function readJson<T>(filePath: string): T | null {
  if (!existsSync(filePath)) return null;
  return JSON.parse(readFileSync(filePath, 'utf-8')) as T;
}

export function getRawDir(date: string): string {
  return join(process.cwd(), 'data', 'raw', date);
}

export function getScoresDir(): string {
  return join(process.cwd(), 'data', 'scores');
}
```

2. Create `src/pipeline/fetchers/inform.ts`:
- Export `async function fetchInform(date: string): Promise<FetchResult>`
- Fetch from INFORM API: `https://drmkc.jrc.ec.europa.eu/inform-index/API/InformAPI/countries/Scores`
- The INFORM API returns JSON with country scores across multiple dimensions
- Parse response to extract: overall INFORM risk score, natural hazard score, health/epidemic score, governance score, climate score per country
- Map indicator names to pipeline indicator names: `inform_health`, `inform_epidemic`, `inform_governance`, `inform_natural`, `inform_climate`
- Write raw response to `data/raw/{date}/inform.json`
- Build `RawSourceData` with `RawIndicator[]` entries for each country
- Write parsed indicators to `data/raw/{date}/inform-parsed.json`
- On fetch failure: log error, return `FetchResult` with `success: false` and error message
- Use Node.js native `fetch()` (available in Node 22)
- If INFORM API is unavailable, try fallback: check for most recent `data/raw/*/inform.json` and use that (log warning about using cached data)

3. Create `src/pipeline/fetchers/gpi.ts`:
- Export `async function fetchGpi(date: string): Promise<FetchResult>`
- GPI data is published annually as an Excel file. The URL may change yearly.
- Attempt to fetch the latest GPI Excel file from Vision of Humanity
- Parse Excel using `xlsx` library to extract: overall GPI score, safety & security domain score, militarisation domain score per country
- Map to indicator names: `gpi_overall`, `gpi_safety_security`, `gpi_militarisation`
- GPI uses country names (not ISO codes) — use the countries.ts mapping to convert names to ISO3 codes
- Write raw file to `data/raw/{date}/gpi.xlsx` and parsed data to `data/raw/{date}/gpi-parsed.json`
- On failure: fall back to cached data from most recent `data/raw/*/gpi-parsed.json`

4. Create `src/pipeline/fetchers/acled.ts`:
- Export `async function fetchAcled(date: string): Promise<FetchResult>`
- ACLED API requires an API key and email (env vars: `ACLED_API_KEY`, `ACLED_EMAIL`)
- If env vars missing, log warning and return FetchResult with success: false, error: "ACLED credentials not configured. Set ACLED_API_KEY and ACLED_EMAIL environment variables."
- Fetch from: `https://api.acleddata.com/acled/read?key={KEY}&email={EMAIL}&event_date={last30days}&event_date_where=BETWEEN&fields=country|event_type|fatalities&limit=0`
- Aggregate by country: count total events and total fatalities in last 30 days
- Map to indicators: `acled_fatalities` (total fatalities), `acled_events` (total events)
- Use country name to ISO3 mapping from countries.ts
- Write raw to `data/raw/{date}/acled.json`, parsed to `data/raw/{date}/acled-parsed.json`
- On failure: fall back to cached data

5. Create `src/pipeline/fetchers/advisories.ts`:
- Export `async function fetchAdvisories(date: string): Promise<FetchResult>`
- US State Dept: Fetch travel advisories from `https://travel.state.gov/content/travel/en/traveladvisories/traveladvisories.html/` or use the data catalog API
- US advisory levels are 1-4 (1=Exercise Normal Precautions, 2=Exercise Increased Caution, 3=Reconsider Travel, 4=Do Not Travel)
- UK FCDO: Fetch from `https://www.gov.uk/api/content/foreign-travel-advice` (JSON API)
- Map US levels to `advisory_level_us` indicator (normalize 1-4 to 0-1 where 1=safe, 4=dangerous: `(level - 1) / 3`)
- Map UK levels similarly to `advisory_level_uk`
- Store both advisory data and normalized indicators
- Write raw to `data/raw/{date}/advisories-us.json` and `data/raw/{date}/advisories-uk.json`
- Parsed combined to `data/raw/{date}/advisories-parsed.json`
- On failure of one source: continue with the other, log warning

- Run all fetchers with `Promise.allSettled()` so one failure doesn't block others
- Return array of FetchResult objects
