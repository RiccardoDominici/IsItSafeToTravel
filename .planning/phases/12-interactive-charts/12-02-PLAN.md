```typescript
function renderTrendChart(): void {
  const section = document.getElementById('trend-overlay')!;
  section.innerHTML = '';
  if (selectedCountries.length === 0) return;

  const countriesWithHistory = selectedCountries.filter(c => {
    const hist = allHistory[c.iso3];
    return hist && hist.length >= 2;
  });

  // ... heading and empty-state logic stays the same ...

  // Store full data for reset
  const fullCountriesWithHistory = countriesWithHistory;

  // Create wrapper elements
  const chartWrapper = document.createElement('div');
  chartWrapper.className = 'relative';

  // Reset button (hidden initially)
  const resetBtn = document.createElement('button');
  resetBtn.className = 'hidden text-xs text-terracotta-600 dark:text-terracotta-400 hover:text-terracotta-800 dark:hover:text-terracotta-200 cursor-pointer mb-1';
  resetBtn.type = 'button';
  resetBtn.textContent = 'Reset';
  section.appendChild(heading);
  section.appendChild(resetBtn);
  section.appendChild(chartWrapper);

  // Inner render function that can be called with filtered date range
  function renderInner(dateRange: [Date, Date] | null): void {
    const width = 800;
    const height = 300;
    const margin = { top: 20, right: 120, bottom: 40, left: 50 };
    const innerWidth = width - margin.left - margin.right;
    const innerHeight = height - margin.top - margin.bottom;

    // Collect all dates, optionally filtered
    const allDates: Date[] = [];
    const filteredHistories: Map<string, HistoryPoint[]> = new Map();

    fullCountriesWithHistory.forEach(c => {
      const hist = allHistory[c.iso3];
      if (!hist) return;
      let filtered = hist;
      if (dateRange) {
        filtered = hist.filter(p => {
          const d = new Date(p.date + 'T00:00:00Z');
          return d >= dateRange[0] && d <= dateRange[1];
        });
      }
      if (filtered.length < 2) filtered = hist; // fallback if range too narrow
      filteredHistories.set(c.iso3, filtered);
      filtered.forEach(p => allDates.push(new Date(p.date + 'T00:00:00Z')));
    });
    allDates.sort((a, b) => a.getTime() - b.getTime());

    if (allDates.length === 0) return;

    const x = scaleTime()
      .domain([allDates[0], allDates[allDates.length - 1]])
      .range([0, innerWidth]);

    const y = scaleLinear()
      .domain([1, 10])
      .range([innerHeight, 0]);

    const lineGen = d3Line<HistoryPoint>()
      .x(d => x(new Date(d.date + 'T00:00:00Z')))
      .y(d => y(d.score));

    // Build SVG content (gridlines, axes, paths, legend — same as current code)
    // ... [keep existing grid/axis/path/legend building logic] ...
    // Add clipPath def:
    // <defs><clipPath id="compare-chart-clip"><rect x="0" y="0" width="${innerWidth}" height="${innerHeight}"/></clipPath></defs>
    // Apply clip-path="url(#compare-chart-clip)" to each country path

    // Build tooltip data from filtered histories
    // ... [keep existing tooltip data building] ...

    chartWrapper.innerHTML = `
      <svg viewBox="0 0 ${width} ${height}" class="w-full text-sand-700 dark:text-sand-300" role="img" aria-label="${t.trend_title}">
        <defs>
          <clipPath id="compare-chart-clip">
            <rect x="0" y="0" width="${innerWidth}" height="${innerHeight}"/>
          </clipPath>
        </defs>
        <g transform="translate(${margin.left}, ${margin.top})">
          ${gridLines}
          ${yLabels}
          ${xLabels}
          <g clip-path="url(#compare-chart-clip)">
            ${paths}
          </g>
        </g>
        <g transform="translate(${margin.left + innerWidth + 15}, ${margin.top})">
          ${legend}
        </g>
      </svg>
      <div id="trend-compare-tooltip" class="absolute hidden pointer-events-none bg-sand-100 dark:bg-sand-800 border border-terracotta-300 dark:border-terracotta-600 rounded px-2 py-1 text-sm shadow-md z-10 whitespace-nowrap"></div>
    `;

    // Re-attach tooltip handlers (same as current code but using filteredHistories for nearest-point lookup)
    const svg = chartWrapper.querySelector('svg')!;
    const tooltip = chartWrapper.querySelector('#trend-compare-tooltip') as HTMLElement;
    // ... [existing tooltip logic, adapted to use tooltipDates from filtered data] ...

    // Attach brush for zoom
    const chartG = svg.querySelector('g')!; // main chart group
    const brush = brushX()
      .extent([[0, 0], [innerWidth, innerHeight]])
      .on('end', (event: any) => {
        if (!event.selection) return;
        const [x0, x1] = event.selection as [number, number];
        const d0 = x.invert(x0);
        const d1 = x.invert(x1);

        // Check if enough data points in range
        let totalPoints = 0;
        fullCountriesWithHistory.forEach(c => {
          const hist = allHistory[c.iso3] || [];
          totalPoints += hist.filter(p => {
            const d = new Date(p.date + 'T00:00:00Z');
            return d >= d0 && d <= d1;
          }).length;
        });
        if (totalPoints < 2) return;

        renderInner([d0, d1]);
        resetBtn.classList.remove('hidden');
      });

    const brushG = document.createElementNS('http://www.w3.org/2000/svg', 'g');
    brushG.setAttribute('class', 'brush');
    chartG.appendChild(brushG);
    select(brushG).call(brush);

    // Style brush selection
    const style = document.createElementNS('http://www.w3.org/2000/svg', 'style');
    style.textContent = '.brush .selection { fill: #c47a5a; fill-opacity: 0.15; stroke: #c47a5a; stroke-opacity: 0.4; }';
    svg.insertBefore(style, svg.firstChild);
  }

  // Reset handler
  resetBtn.addEventListener('click', () => {
    renderInner(null);
    resetBtn.classList.add('hidden');
  });

  // Initial render
  renderInner(null);
}
```

**Key implementation details:**
- Wrap country line paths in a `<g clip-path="url(#compare-chart-clip)">` for clean zoom clipping
- Use clipPath id "compare-chart-clip" (different from TrendChart's "chart-clip" to avoid conflicts if both on same page — though they aren't currently)
- When zooming, filter ALL countries' history to the same date range for consistent x-axis
- If a country has fewer than 2 points in the zoomed range, fall back to its full history
- Tooltip must show only the filtered data points, not all historical data
- Brush overlay `.selection` styled with semi-transparent terracotta (#c47a5a) at 0.15 opacity
- Use 300ms easeCubicOut transitions on path elements when zooming/resetting (apply via `select(pathEl).transition().duration(300).ease(easeCubicOut)`)
- ALL date parsing uses `new Date(d.date + 'T00:00:00Z')` (Phase 11 fix, non-negotiable)
- ALL axis labels use `utcFormat('%b %d')` (Phase 11 fix, non-negotiable)
- Brush is re-attached after every renderInner call since innerHTML replaces SVG content
