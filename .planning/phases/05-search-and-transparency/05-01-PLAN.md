Script section (`<script>` tag, NOT `is:inline` -- Astro bundles and deduplicates automatically):
```typescript
import Fuse from 'fuse.js';

interface SearchItem {
  iso3: string;
  name: string;
  score: number | null;
  color: string | null;
}

const container = document.querySelector('[data-search]') as HTMLElement;
if (container) {
  const searchData: SearchItem[] = JSON.parse(container.dataset.search!);
  const lang = container.dataset.lang!;
  const routeSegment = container.dataset.routeSegment!;
  const noResultsText = container.dataset.noResults!;

  const fuse = new Fuse(searchData, {
    keys: ['name'],
    threshold: 0.3,
    minMatchCharLength: 2,
    includeScore: true,
    limit: 10,
  });

  const toggle = document.getElementById('search-toggle')!;
  const panel = document.getElementById('search-panel')!;
  const input = document.getElementById('search-input') as HTMLInputElement;
  const listbox = document.getElementById('search-listbox')!;
  let activeIndex = -1;
  let currentResults: SearchItem[] = [];

  function open() {
    panel.classList.remove('hidden');
    toggle.setAttribute('aria-expanded', 'true');
    panel.setAttribute('aria-expanded', 'true');
    input.focus();
  }

  function close() {
    panel.classList.add('hidden');
    toggle.setAttribute('aria-expanded', 'false');
    panel.setAttribute('aria-expanded', 'false');
    input.value = '';
    listbox.innerHTML = '';
    activeIndex = -1;
    currentResults = [];
    toggle.focus();
  }

  function navigate(iso3: string) {
    window.location.href = `/${lang}/${routeSegment}/${iso3.toLowerCase()}/`;
  }

  function renderResults(items: SearchItem[]) {
    currentResults = items;
    activeIndex = -1;
    if (items.length === 0 && input.value.length >= 2) {
      listbox.innerHTML = `<li class="px-4 py-3 text-sm text-sand-500 dark:text-sand-400">${noResultsText}</li>`;
      return;
    }
    listbox.innerHTML = items.map((item, i) => `
      <li role="option" id="result-${i}" aria-selected="false"
          data-iso3="${item.iso3}"
          class="px-4 py-2 cursor-pointer flex items-center justify-between hover:bg-sand-100 dark:hover:bg-sand-700 text-sm text-sand-800 dark:text-sand-100">
        <span>${item.name}</span>
        ${item.score !== null ? `<span class="inline-flex items-center gap-1 text-xs font-medium"><span class="w-2.5 h-2.5 rounded-full" style="background:${item.color}"></span>${item.score}/10</span>` : ''}
      </li>
    `).join('');
    // Click handlers on results
    listbox.querySelectorAll('[data-iso3]').forEach(el => {
      el.addEventListener('click', () => navigate(el.getAttribute('data-iso3')!));
    });
  }

  function setActive(index: number) {
    listbox.querySelectorAll('[role="option"]').forEach((el, i) => {
      el.setAttribute('aria-selected', i === index ? 'true' : 'false');
      el.classList.toggle('bg-sand-100', i === index);
      el.classList.toggle('dark:bg-sand-700', i === index);
    });
    activeIndex = index;
    input.setAttribute('aria-activedescendant', index >= 0 ? `result-${index}` : '');
  }

  toggle.addEventListener('click', () => {
    panel.classList.contains('hidden') ? open() : close();
  });

  input.addEventListener('input', () => {
    const query = input.value.trim();
    if (query.length < 2) { listbox.innerHTML = ''; currentResults = []; return; }
    const results = fuse.search(query).map(r => r.item);
    renderResults(results);
  });

  input.addEventListener('keydown', (e) => {
    if (e.key === 'Escape') { close(); return; }
    if (e.key === 'ArrowDown') {
      e.preventDefault();
      if (currentResults.length > 0) setActive(Math.min(activeIndex + 1, currentResults.length - 1));
    }
    if (e.key === 'ArrowUp') {
      e.preventDefault();
      if (currentResults.length > 0) setActive(Math.max(activeIndex - 1, 0));
    }
    if (e.key === 'Enter' && activeIndex >= 0 && currentResults[activeIndex]) {
      navigate(currentResults[activeIndex].iso3);
    }
  });

  // Close on click outside
  document.addEventListener('click', (e) => {
    if (!container.contains(e.target as Node)) close();
  });
}
```

- Build-time data serialization via data-attributes (matches map component pattern)
- Fuse.js initialized once on the 248-item dataset
- ARIA combobox pattern for accessibility
- CSS-only show/hide via `hidden` class toggle
- Score color dots use `scoreToColor()` output serialized at build time
- Navigation uses `window.location.href` to `/{lang}/{routeSegment}/{iso3.toLowerCase()}/`
